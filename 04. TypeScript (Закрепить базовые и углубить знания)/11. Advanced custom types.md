# Pick

```
type MyPick<T, K extends keyof T> = {
    [U in K]: T[U];
}
type X = { a: string, b: number, c: boolean };
type PickedX = MyPick<X, 'a' | 'c'>;
const x: PickedX = { a: 'sdsds', c: true };
```

# Partial

```
type MyPartial<T> = {
    [K in keyof T]?: T[K];
}
```

# ReturnType

```
type MyReturnType<T> = T extends (...args: any[]) => infer R ? R : never;
function func(): string {
    return '';
}
type ReturnAType = MyReturnType<typeof func>; // string
```
